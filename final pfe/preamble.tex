% ─── CUSTOM PHASE BLOCKS ────────────────────────────────────────────────────
% Two callable forms are provided so that all three source reports compile
% unchanged:
%   * \phaseblock{N}{Title}{content}    (used by tahkik_pfe and model train.tex)
%   * \phaseblocks{Title}{content}      (used by server_report*.tex; the original
%                                        files write the literal "Phase X --- Y"
%                                        title themselves in argument 1)
\newcommand{\phaseblock}[3]{%
  \begin{tcolorbox}[
    enhanced, colback=white, colframe=black,
    leftrule=5pt, sharp corners, before upper={\parindent0pt}
  ]
  {\bfseries\color{black}Phase #1 --- #2}\\[4pt]
  \small #3
  \end{tcolorbox}%
}

\newcommand{\phaseblocks}[2]{%
  \begin{tcolorbox}[
    enhanced, colback=white, colframe=black,
    leftrule=5pt, sharp corners, before upper={\parindent0pt}
  ]
  {\bfseries\color{black}#1}\\[4pt]
  \small #2
  \end{tcolorbox}%
}
